using System;
using System.IO;
using System.Collections.Generic;
using System.Runtime.InteropServices; // 用于原生键盘监听
using BepInEx;
using BepInEx.Unity.IL2CPP;
using UnityEngine;

namespace shiyi_mod
{
    [BepInPlugin("com.shiyi.tool.extractor", "Shiyi.SpriteExtractor", "1.1.0")]
    public class SpriteExtractorPlugin : BasePlugin
    {
        public override void Load()
        {
            System.Console.OutputEncoding = System.Text.Encoding.UTF8;
            Log.LogInfo("[Shiyi] 提取器已启动！进游戏看到画面后，按 F10 键开始偷图！");
            AddComponent<SpriteDumper>();
        }
    }

    public class SpriteDumper : MonoBehaviour
    {
        // 修复 1: 换回原生的 Windows API 监听，彻底抛弃 UnityEngine.Input
        [DllImport("user32.dll")]
        public static extern short GetAsyncKeyState(int vKey);
        private const int VK_F10 = 0x79; // F10 键的 Windows 键码
        private bool _lastKeyState = false;

        public static string DumpFolder = Path.Combine(Paths.PluginPath, "ShiyiCustomTexture", "SpriteDump");

        void Update()
        {
            // 原生按键检测
            bool isDown = (GetAsyncKeyState(VK_F10) & 0x8000) != 0;
            if (isDown && !_lastKeyState)
            {
                ExtractAllSprites();
            }
            _lastKeyState = isDown;
        }

        public static void ExtractAllSprites()
        {
            if (!Directory.Exists(DumpFolder)) Directory.CreateDirectory(DumpFolder);

            System.Console.WriteLine("[Shiyi-解包器] 开始抓取当前内存中的贴图...");

            // 修复 2: 删除了不支持泛型的 LoadAll，直接抓取当前屏幕和内存中已经存在的资源
            var allSprites = Resources.FindObjectsOfTypeAll<Sprite>();
            int successCount = 0;
            HashSet<string> savedNames = new HashSet<string>();

            foreach (var sprite in allSprites)
            {
                if (sprite == null || sprite.texture == null) continue;

                // 过滤掉太小的无用像素点
                if (sprite.rect.width < 10 || sprite.rect.height < 10) continue;

                string baseName = string.IsNullOrEmpty(sprite.name) ? "unnamed_sprite" : sprite.name;
                foreach (var c in Path.GetInvalidFileNameChars()) { baseName = baseName.Replace(c, '_'); }

                string saveName = baseName;
                int suffix = 1;
                while (savedNames.Contains(saveName))
                {
                    saveName = $"{baseName}_{suffix}";
                    suffix++;
                }
                savedNames.Add(saveName);

                string filePath = Path.Combine(DumpFolder, saveName + ".png");

                try
                {
                    Texture2D croppedTex = CropSpriteToTexture(sprite);
                    
                    if (croppedTex != null)
                    {
                        // 修复 3: IL2CPP 环境下，不用 ImageConversion，直接使用 Texture2D 的扩展方法
                        byte[] bytes = croppedTex.EncodeToPNG();
                        File.WriteAllBytes(filePath, bytes);
                        
                        UnityEngine.Object.Destroy(croppedTex); // 释放内存
                        successCount++;
                    }
                }
                catch 
                {
                    // 忽略切割失败的报错
                }
            }

            System.Console.WriteLine($"[Shiyi-解包器] 提取完成！成功保存了 {successCount} 个元素！");
            System.Console.WriteLine($"[Shiyi-解包器] 路径: {DumpFolder}");
        }

        private static Texture2D CropSpriteToTexture(Sprite sprite)
        {
            Texture2D sourceTex = sprite.texture;
            Rect rect = sprite.rect; 

            RenderTexture tmp = RenderTexture.GetTemporary(
                sourceTex.width, 
                sourceTex.height, 
                0, 
                RenderTextureFormat.Default, 
                RenderTextureReadWrite.Linear);
                
            Graphics.Blit(sourceTex, tmp);
            
            RenderTexture previous = RenderTexture.active;
            RenderTexture.active = tmp;
            
            Texture2D croppedTex = new Texture2D((int)rect.width, (int)rect.height, TextureFormat.RGBA32, false);
            
            croppedTex.ReadPixels(new Rect(rect.x, rect.y, rect.width, rect.height), 0, 0);
            croppedTex.Apply();
            
            RenderTexture.active = previous;
            RenderTexture.ReleaseTemporary(tmp);
            
            return croppedTex;
        }
    }
}

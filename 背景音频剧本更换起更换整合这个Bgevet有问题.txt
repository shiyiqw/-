        public override void Load()
        {
            Console.OutputEncoding = Encoding.UTF8;
            Console.InputEncoding = Encoding.UTF8;

            DumpFilePath = Path.Combine(Paths.PluginPath, "ScenarioDump.csv");
            LoadScenarioDump();
            CustomVoiceManager.Initialize();
            ExternalBackgroundManager.Initialize();
            AddComponent<ExternalBackgroundUpdateHandler>();

            Harmony harmony = new Harmony("com.shiyimv.utage_csv_injector");
            harmony.PatchAll();
            PatchBackgroundCommandConstructors(harmony);
            PatchBackgroundCommands(harmony);

            UnityEngine.Debug.Log($"[Mod] 剧本注入 + 自定义人物语音 + 外部背景 已加载 | ver=1.8.1(rowindex-native-swap) | bgPlaceholder=[{ExternalBackgroundPlaceholder}]");
        }

        public static void LoadScenarioDump()
        {
            try
            {
                UnityEngine.Debug.Log($"[Mod-Debug] 开始读取 ScenarioDump: {DumpFilePath}");
                if (!File.Exists(DumpFilePath))
                {
                    UnityEngine.Debug.LogWarning($"[Mod-Debug] ScenarioDump 不存在: {DumpFilePath}");
                    return;
                }
                CustomSheetsCache.Clear();
                ExternalBackgroundRowMap.Clear();

                string currentSheetName = null;
                List<string[]> currentRows = new List<string[]>();
                string[] lines = File.ReadAllLines(DumpFilePath, Encoding.UTF8);

                foreach (var line in lines)
                {
                    if (string.IsNullOrEmpty(line) || line.Trim().Length == 0) continue;

                    if (line.StartsWith("#SHEET_NAME", StringComparison.OrdinalIgnoreCase))
                    {
                        if (!string.IsNullOrEmpty(currentSheetName) && currentRows.Count > 0)
                        {
                            CustomSheetsCache[currentSheetName] = new List<string[]>(currentRows);
                        }

                        string[] parts = line.Split(',');
                        if (parts.Length > 1)
                        {
                            currentSheetName = parts[1].Trim();
                            currentRows.Clear();
                        }
                        continue;
                    }

                    if (!string.IsNullOrEmpty(currentSheetName))
                    {
                        string[] cells = line.Split(',');
                        int rowIndex = currentRows.Count;
                        RewriteExternalVoiceCells(cells);
                        RewriteExternalBackgroundCells(currentSheetName, rowIndex, cells);
                        currentRows.Add(cells);

                        if (ContainsExternalVoiceMarker(cells))
                        {
                            UnityEngine.Debug.Log($"[Mod-Debug] 缓存到外部语音行: sheet=[{currentSheetName}] raw=[{string.Join("|", cells)}]");
                        }
                    }
                }

                if (!string.IsNullOrEmpty(currentSheetName) && currentRows.Count > 0)
                {
                    CustomSheetsCache[currentSheetName] = new List<string[]>(currentRows);
                }

                UnityEngine.Debug.Log($"[Mod] 成功载入自定义剧本，共成功缓存了 {CustomSheetsCache.Count} 张工作表。");
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[Mod] 载入剧本文件失败: {ex.Message}");
            }
        }

        private static void RewriteExternalVoiceCells(string[] cells)
        {
            try
            {
                if (cells == null || cells.Length == 0)
                {
                    UnityEngine.Debug.LogWarning("[Mod-Debug] RewriteExternalVoiceCells 收到空行");
                    return;
                }

                int voiceIndex = -1;
                for (int i = 0; i < cells.Length; i++)
                {
                    if (string.Equals(cells[i]?.Trim(), "Voice", StringComparison.OrdinalIgnoreCase))
                    {
                        voiceIndex = i;
                        break;
                    }
                }

                if (voiceIndex < 0)
                {
                    return;
                }

                for (int i = voiceIndex + 1; i < cells.Length; i++)
                {
                    string fileCell = cells[i]?.Trim();
                    if (!CustomVoiceManager.TryParseExternalVoicePath(fileCell, out var actualPath))
                    {
                        continue;
                    }

                    UnityEngine.Debug.Log($"[Mod-Debug] 命中外部语音单元格: index={i} value=[{fileCell}] parsed=[{actualPath}]");

                    string token = $"__extvoice_{ExternalVoiceId++}";
                    CustomVoiceManager.RegisterExternalVoiceToken(token, actualPath);
                    cells[i] = token;

                    UnityEngine.Debug.Log($"[CustomVoice] 剧本外部语音已改写: index={i} {fileCell} -> {token}");
                    return;
                }
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[Mod] 载入剧本文件失败: {ex.Message}");
            }
        }

        private static void RewriteExternalBackgroundCells(string sheetName, int rowIndex, string[] cells)
        {
            try
            {
                if (cells == null || cells.Length < 2)
                {
                    return;
                }

                for (int i = 0; i < cells.Length - 1; i++)
                {
                    string cmd = cells[i]?.Trim();
                    if (!string.Equals(cmd, "Bg", StringComparison.OrdinalIgnoreCase))
                    {
                        continue;
                    }

                    string bgValue = cells[i + 1]?.Trim();
                    if (!CustomVoiceManager.TryParseExternalVoicePath(bgValue, out _))
                    {
                        continue;
                    }

                    string key = BuildRowKey(sheetName, rowIndex);
                    ExternalBackgroundRowMap[key] = bgValue;
                    cells[i + 1] = ExternalBackgroundPlaceholder;
                    UnityEngine.Debug.Log($"[CustomBG] 载入阶段已改写背景占位: sheet=[{sheetName}] row=[{rowIndex}] col=[{i + 1}] raw=[{bgValue}] placeholder=[{ExternalBackgroundPlaceholder}]");
                    return;
                }
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] 载入阶段改写背景失败: {ex.Message}");
            }
        }

        private static bool ContainsExternalVoiceMarker(string[] cells)
        {
            if (cells == null) return false;

            foreach (var cell in cells)
            {
                if (CustomVoiceManager.IsExternalVoicePath(cell))
                {
                    return true;
                }
            }

            return false;
        }

        {
            TryPatchCtor(harmony, "Utage.AdvCommandBg", nameof(AdvCommandBg_Ctor_Prefix));
            TryPatchCtor(harmony, "Utage.AdvCommandBgEvent", nameof(AdvCommandBg_Ctor_Prefix));
            TryPatchCtor(harmony, "Utage.AdvCommandBgBase", nameof(AdvCommandBg_Ctor_Prefix));
        }

        {
            TryPatchDoCommand(harmony, "Utage.AdvCommandBg", nameof(AdvCommandBgBase_DoCommand_Prefix));
            TryPatchDoCommand(harmony, "Utage.AdvCommandBgEvent", nameof(AdvCommandBgBase_DoCommand_Prefix));
            TryPatchDoCommand(harmony, "Utage.AdvCommandBgBase", nameof(AdvCommandBgBase_DoCommand_Prefix));

            TryPatchDoCommand(harmony, "Utage.AdvCommandBgOff", nameof(AdvCommandBgOff_DoCommand_Prefix));
            TryPatchDoCommand(harmony, "Utage.AdvCommandBgEventOff", nameof(AdvCommandBgOff_DoCommand_Prefix));
            TryPatchDoCommand(harmony, "Utage.AdvCommandBgOffOnly", nameof(AdvCommandBgOff_DoCommand_Prefix));
            TryPatchDoCommand(harmony, "Utage.AdvCommandBgOffBase", nameof(AdvCommandBgOff_DoCommand_Prefix));
        }

        private static void TryPatchCtor(Harmony harmony, string typeName, string prefixMethodName)
        {
            try
            {
                var targetType = AccessTools.TypeByName(typeName);
                var targetMethod = targetType == null ? null : AccessTools.Constructor(targetType, new Type[] { typeof(StringGridRow), typeof(AdvSettingDataManager) });
                var prefixMethod = AccessTools.Method(typeof(Plugin), prefixMethodName);
                if (targetMethod == null || prefixMethod == null)
                {
                    UnityEngine.Debug.LogWarning($"[CustomBG] 构造补丁失败: type=[{typeName}] target=[{targetMethod != null}] prefix=[{prefixMethod != null}]");
                    return;
                }

                harmony.Patch(targetMethod, prefix: new HarmonyMethod(prefixMethod));
                UnityEngine.Debug.Log($"[CustomBG] 已挂钩背景构造: {typeName}..ctor");
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] 挂钩背景构造失败: {typeName} | {ex.Message}");
            }
        }

        private static void TryPatchDoCommand(Harmony harmony, string typeName, string prefixMethodName)
        {
            try
            {
                var targetType = AccessTools.TypeByName(typeName);
                var targetMethod = targetType == null ? null : AccessTools.Method(targetType, "DoCommand", new Type[] { typeof(AdvEngine) });
                var prefixMethod = AccessTools.Method(typeof(Plugin), prefixMethodName);
                if (targetMethod == null || prefixMethod == null)
                {
                    UnityEngine.Debug.LogWarning($"[CustomBG] 反射补丁失败: type=[{typeName}] target=[{targetMethod != null}] prefix=[{prefixMethod != null}]");

                harmony.Patch(targetMethod, prefix: new HarmonyMethod(prefixMethod));

            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] 挂钩背景命令失败: {typeName} | {ex.Message}");
            }
        }

        private static void AdvCommandBg_Ctor_Prefix(StringGridRow row)

        private static bool AdvCommandBgBase_DoCommand_Prefix(object __instance, AdvEngine engine)
        {
            try
            {
                var graphicList = GetTraverseFieldValue(__instance, "graphic", default(AdvGraphicInfoList));
                string label = GetTraverseFieldValue(__instance, "label", string.Empty);
                float fadeTime = GetTraverseFieldValue(__instance, "fadeTime", 0f);
                string fileName = GetBackgroundGraphicFileName(graphicList);
                string rawBgValue = GetExternalBackgroundPathFromCommand(__instance);

                if (string.IsNullOrWhiteSpace(rawBgValue))
                {
                    rawBgValue = !string.IsNullOrWhiteSpace(fileName) ? fileName : label;
                }

                if (!CustomVoiceManager.TryParseExternalVoicePath(rawBgValue, out var actualPath))
                {
                    ExternalBackgroundManager.ClearActiveExternalBackground();
                    return true;
                }

                if (string.IsNullOrWhiteSpace(fileName))
                {
                    UnityEngine.Debug.Log($"[CustomBG] graphic.FileName 为空，回退使用外部映射/label 解析背景: [{rawBgValue}]");
                }

                fileName = rawBgValue;

                ExternalBackgroundManager.SetActiveExternalBackground(actualPath, fadeTime);
                UnityEngine.Debug.Log($"[CustomBG] 原生背景劫持已就绪: label=[{label}] raw=[{fileName}] path=[{actualPath}] fade=[{fadeTime}]");
                return true;
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] 处理背景命令失败: {ex.Message}");
                return true;
            }
        }

        private static void AdvCommandBgOff_DoCommand_Prefix(object __instance, AdvEngine engine)
        {
            ExternalBackgroundManager.ClearActiveExternalBackground();
        }

        private static T GetTraverseFieldValue<T>(object instance, string fieldName, T defaultValue)
        {
            try
            {
                if (instance == null) return defaultValue;
                var traverse = Traverse.Create(instance).Field(fieldName);
                if (traverse == null) return defaultValue;
                return traverse.GetValue<T>();
            }
            catch
            {
                return defaultValue;
            }
        }

        private static string GetBackgroundGraphicFileName(AdvGraphicInfoList graphicList)
        {
            try
            {
                if (graphicList == null) return null;

                if (graphicList.Main != null && !string.IsNullOrWhiteSpace(graphicList.Main.FileName))
                {
                    return graphicList.Main.FileName;
                }

                if (graphicList.InfoList != null)
                {
                    foreach (var info in graphicList.InfoList)
                    {
                        if (info != null && !string.IsNullOrWhiteSpace(info.FileName))
                        {
                            return info.FileName;
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] 读取背景文件名失败: {ex.Message}");
            }

            return null;
        }

        private static string GetExternalBackgroundPathFromCommand(object instance)
        {
            try
            {
                if (instance == null) return null;

                object rowDataObj = Traverse.Create(instance).Property("RowData").GetValue();
                if (rowDataObj is StringGridRow row)
                {
                    string key = BuildRowKey(row.Grid?.SheetName, row.RowIndex);
                    if (ExternalBackgroundRowMap.TryGetValue(key, out var rawBg))
                    {
                        return rawBg;
                    }
                }
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogWarning($"[CustomBG] 读取命令行对应的外部背景映射失败: {ex.Message}");
            }

            return null;
        }

        internal static bool TryRewriteExternalBackgroundRow(StringGridRow row)
        {
            try
            {
                if (row == null || row.Grid == null) return false;

                string sheetName = row.Grid.SheetName ?? string.Empty;
                int rowIndex = row.RowIndex;
                string key = BuildRowKey(sheetName, rowIndex);
                if (!ExternalBackgroundRowMap.TryGetValue(key, out var bgValue))
                {
                    return false;
                }

                var customRows = CustomSheetsCache.TryGetValue(sheetName, out var rows) ? rows : null;
                if (customRows == null || rowIndex < 0 || rowIndex >= customRows.Count)
                {
                    return false;
                }

                var rowData = customRows[rowIndex];
                if (rowData == null || rowData.Length < 2)
                {
                    return false;
                }

                for (int i = 0; i < rowData.Length - 1; i++)
                {
                    if (!string.Equals(rowData[i]?.Trim(), "Bg", StringComparison.OrdinalIgnoreCase))
                    {
                        continue;
                    }

                    if (!string.Equals(rowData[i + 1], ExternalBackgroundPlaceholder, StringComparison.OrdinalIgnoreCase))
                    {
                        rowData[i + 1] = ExternalBackgroundPlaceholder;
                    }

                    var il2cppArray = new Il2CppInterop.Runtime.InteropTypes.Arrays.Il2CppStringArray(rowData);
                    row.InitFromStringArray(il2cppArray);
                    UnityEngine.Debug.Log($"[CustomBG] 构造前已替换背景占位: sheet=[{sheetName}] row=[{rowIndex}] col=[{i + 1}] raw=[{bgValue}] placeholder=[{ExternalBackgroundPlaceholder}]");
                    return true;
                }
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] 构造前替换背景占位失败: {ex.Message}");
            }

            return false;
        }

        private static string BuildRowKey(string sheetName, int rowIndex)
        {
            return $"{sheetName ?? string.Empty}::{rowIndex}";
        }
    }

    public static class CustomVoiceManager
    {
        private const string ExternalVoicePrefix = "ext:";
        private const string ExternalVoicePrefixAlias = "ent:";
        private const string LegacyLocalVoicePrefix = "L:";
        private const string ExternalVoiceTokenPrefix = "__extvoice_";
        private static readonly StringComparison PathComparison = StringComparison.OrdinalIgnoreCase;
        private static readonly Dictionary<string, string> ExternalVoiceTokenMap = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
        private static readonly Dictionary<string, AudioClip> ClipCache = new Dictionary<string, AudioClip>(StringComparer.OrdinalIgnoreCase);
        private static readonly HashSet<string> MissingPathLogged = new HashSet<string>(StringComparer.OrdinalIgnoreCase);

        public static void Initialize()
        {
            UnityEngine.Debug.Log("[CustomVoice] 仅当前缀为 ent:/ext:/L: 时才会替换语音");
            UnityEngine.Debug.Log("[CustomVoice] 剧本可直接写法: ,Voice,角色名,ent:相对或绝对wav路径");
            UnityEngine.Debug.Log("[CustomVoice] 示例1: ,Voice,任意角色,ent:ShiyiAssets_List\\Voice\\林翩翩\\诅咒方知宥.wav");
            UnityEngine.Debug.Log("[CustomVoice] 示例2: ,Voice,任意角色,L:ShiyiAssets_List\\Voice\\林翩翩\\诅咒方知宥.wav");
        }

        public static void RegisterExternalVoiceToken(string token, string actualPath)
        {
            if (string.IsNullOrWhiteSpace(token) || string.IsNullOrWhiteSpace(actualPath)) return;
            ExternalVoiceTokenMap[token.Trim()] = actualPath.Trim();
        }

        public static bool TryGetClip(string characterLabel, AssetFile assetFile, out AudioClip clip)
        {
            clip = null;

            string fileName = GetAssetFileNameSafe(assetFile);
            bool hasTokenRequest = IsExternalVoiceToken(fileName);
            bool hasDirectRequest = IsExternalVoiceFileName(fileName);
            bool hasExternalRequest = hasTokenRequest || hasDirectRequest;

            UnityEngine.Debug.Log($"[Mod-Debug] TryGetClip: role=[{characterLabel ?? "<null>"}] file=[{fileName ?? "<null>"}] token={hasTokenRequest} direct={hasDirectRequest}");

            if (!hasExternalRequest)
            {
                return false;
            }

            foreach (var candidate in BuildCandidatePaths(fileName))
            {
                if (TryLoadClip(candidate, out clip))
                {
                    UnityEngine.Debug.Log($"[CustomVoice] 命中外部语音: role=[{characterLabel}] file=[{fileName}] path=[{candidate}]");
                    return true;
                }
            }

            return false;
        }

        public static string GetAssetFileNameSafe(object assetFile)
        {
            try
            {
                if (assetFile is AssetFile typedAssetFile)
                {
                    return typedAssetFile.FileName;
                }

                if (assetFile is AssetFileBase typedAssetFileBase)
                {
                    return typedAssetFileBase.FileName;
                }

                return null;
            }
            catch
            {
                return null;
            }
        }

        public static bool IsExternalVoiceFileName(string fileName)
        {
            return TryParseExternalVoicePath(fileName, out _);
        }

        public static bool IsExternalVoiceToken(string fileName)
        {
            return !string.IsNullOrEmpty(fileName) && fileName.StartsWith(ExternalVoiceTokenPrefix, PathComparison);
        }

        public static bool IsExternalVoicePath(string loadPath)
        {
            return TryParseExternalVoicePath(loadPath, out _);
        }

        public static bool TryParseExternalVoicePath(string rawValue, out string actualPath)
        {
            actualPath = null;
            if (string.IsNullOrWhiteSpace(rawValue)) return false;

            string value = rawValue.Trim();
            int prefixIndex = FindExternalVoicePrefixIndex(value, out string prefix);
            if (prefixIndex < 0 || string.IsNullOrEmpty(prefix)) return false;

            actualPath = value.Substring(prefixIndex + prefix.Length).Trim().TrimStart('\\', '/');
            return !string.IsNullOrEmpty(actualPath);
        }

        private static int FindExternalVoicePrefixIndex(string value, out string prefix)
        {
            prefix = null;
            if (string.IsNullOrEmpty(value)) return -1;

            int extIndex = value.IndexOf(ExternalVoicePrefix, PathComparison);
            int entIndex = value.IndexOf(ExternalVoicePrefixAlias, PathComparison);
            int legacyIndex = value.IndexOf(LegacyLocalVoicePrefix, PathComparison);

            int bestIndex = int.MaxValue;

            if (extIndex >= 0 && extIndex < bestIndex)
            {
                bestIndex = extIndex;
                prefix = ExternalVoicePrefix;
            }
            if (entIndex >= 0 && entIndex < bestIndex)
            {
                bestIndex = entIndex;
                prefix = ExternalVoicePrefixAlias;
            }
            if (legacyIndex >= 0 && legacyIndex < bestIndex)
            {
                bestIndex = legacyIndex;
                prefix = LegacyLocalVoicePrefix;
            }

            return bestIndex == int.MaxValue ? -1 : bestIndex;
        }

        private static IEnumerable<string> BuildCandidatePaths(string fileName)
        {
            if (string.IsNullOrEmpty(fileName))
            {
                yield break;
            }

            if (ExternalVoiceTokenMap.TryGetValue(fileName.Trim(), out var tokenPath))
            {
                UnityEngine.Debug.Log($"[Mod-Debug] BuildCandidatePaths 命中 token: [{fileName}] -> [{tokenPath}]");
                yield return tokenPath;
                if (!Path.HasExtension(tokenPath))
                {
                    yield return tokenPath + ".wav";
                }
                yield break;
            }

            if (TryParseExternalVoicePath(fileName, out var externalPath))
            {
                UnityEngine.Debug.Log($"[Mod-Debug] BuildCandidatePaths 命中直接前缀路径: [{fileName}] -> [{externalPath}]");
                yield return externalPath;
                if (!Path.HasExtension(externalPath))
                {
                    yield return externalPath + ".wav";
                }

                if (fileName.IndexOf(LegacyLocalVoicePrefix, PathComparison) >= 0)
                {
                    foreach (var legacyCandidate in BuildLegacyLocalCandidates(externalPath))
                    {
                        yield return legacyCandidate;
                    }
                }
            }
        }

        private static IEnumerable<string> BuildLegacyLocalCandidates(string externalPath)
        {
            if (string.IsNullOrWhiteSpace(externalPath))
            {
                yield break;
            }

            string normalized = externalPath.Trim().TrimStart('\\', '/').Replace('/', '\\');
            if (!string.IsNullOrEmpty(normalized))
            {
                yield return Path.Combine("ShiyiAssets_List", "Voice", normalized);
            }

            string fileNameOnly = Path.GetFileName(normalized);
            if (!string.IsNullOrEmpty(fileNameOnly))
            {
                yield return Path.Combine("ShiyiAssets_List", "Voice", fileNameOnly);
            }
        }

        private static bool TryLoadClip(string pathOrRelativePath, out AudioClip clip)
        {
            clip = null;
            if (string.IsNullOrEmpty(pathOrRelativePath)) return false;

            if (ClipCache.TryGetValue(pathOrRelativePath, out clip) && clip != null)
            {
                return true;
            }

            string fullPath = Path.IsPathRooted(pathOrRelativePath)
                ? pathOrRelativePath
                : Path.Combine(Paths.PluginPath, pathOrRelativePath);

            if (!File.Exists(fullPath))
            {
                if (MissingPathLogged.Add(fullPath))
                {
                    UnityEngine.Debug.LogWarning($"[CustomVoice] 未找到候选语音: {fullPath}");
                }
                return false;
            }

            try
            {
                clip = LoadWavAsAudioClip(fullPath);
                if (clip == null) return false;

                ClipCache[pathOrRelativePath] = clip;
                UnityEngine.Debug.Log($"[CustomVoice] 已加载语音: {fullPath}");
                return true;
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomVoice] 读取语音失败: {fullPath} | {ex.Message}");
                return false;
            }
        }

        private static AudioClip LoadWavAsAudioClip(string filePath)
        {
            byte[] bytes = File.ReadAllBytes(filePath);
            if (bytes.Length < 44) throw new Exception("WAV 文件长度异常，无法解析");

            int channels = BitConverter.ToInt16(bytes, 22);
            int frequency = BitConverter.ToInt32(bytes, 24);
            int bitsPerSample = BitConverter.ToInt16(bytes, 34);
            if (bitsPerSample != 16) throw new Exception($"当前仅支持 16-bit PCM WAV，检测到 {bitsPerSample}-bit");

            int pos = 12;
            while (pos + 8 < bytes.Length)
            {
                bool isDataChunk = bytes[pos] == 100 && bytes[pos + 1] == 97 && bytes[pos + 2] == 116 && bytes[pos + 3] == 97;
                if (isDataChunk) break;

                int chunkSize = BitConverter.ToInt32(bytes, pos + 4);
                pos += 8 + chunkSize;
            }

            if (pos + 8 >= bytes.Length) throw new Exception("WAV 中未找到 data 数据块");

            pos += 4;
            int dataSize = BitConverter.ToInt32(bytes, pos);
            pos += 4;

            int sampleCount = dataSize / 2;
            float[] wavData = new float[sampleCount];
            for (int i = 0; i < sampleCount; i++)
            {
                wavData[i] = BitConverter.ToInt16(bytes, pos + i * 2) / 32768.0f;
            }

            AudioClip clip = AudioClip.Create(Path.GetFileNameWithoutExtension(filePath), sampleCount / channels, channels, frequency, false);
            clip.SetData(wavData, 0);
            return clip;
        }
    }

    public static class ExternalBackgroundManager
    {
        private static readonly Dictionary<string, Texture2D> TextureCache = new Dictionary<string, Texture2D>(StringComparer.OrdinalIgnoreCase);
        private static readonly Dictionary<string, Sprite> SpriteCache = new Dictionary<string, Sprite>(StringComparer.OrdinalIgnoreCase);
        private static readonly HashSet<string> MissingPathLogged = new HashSet<string>(StringComparer.OrdinalIgnoreCase);
        private static readonly string[] ExtraExtensions = new[] { ".png", ".jpg", ".jpeg", ".bmp" };
        private static string _activePath;
        private static string _lastAppliedPath;

        public static void Initialize()
        {
            UnityEngine.Debug.Log("[CustomBG] 已启用外部背景读取（RowIndex + 原生层替换模式），剧本可直接写: Bg,ent:Picture/go.png");
            UnityEngine.Debug.Log("[CustomBG] 相对路径基于 BepInEx/plugins 目录解析");
        }

        public static void SetActiveExternalBackground(string actualPath, float fadeTime)
        {
            if (string.IsNullOrWhiteSpace(actualPath))
            {
                ClearActiveExternalBackground();
                return;
            }

            string normalizedPath = NormalizePath(actualPath);
            if (string.IsNullOrEmpty(normalizedPath))
            {
                ClearActiveExternalBackground();
                return;
            }

            _activePath = normalizedPath;
            PreloadTexture(normalizedPath);
            PreloadSprite(normalizedPath);
            UnityEngine.Debug.Log($"[CustomBG] 已登记 RowIndex 背景替换目标: path=[{normalizedPath}] fade=[{fadeTime}]");
        }

        public static void ClearActiveExternalBackground()
        {
            _activePath = null;
            _lastAppliedPath = null;
        }

        public static void UpdateTick()
        {
            try
            {
                if (string.IsNullOrEmpty(_activePath))
                {
                    return;
                }

                GameObject bgContainer = GameObject.Find("AdvEngine/GraphicManager/BG");
                if (bgContainer == null)
                {
                    bgContainer = GameObject.Find("GraphicManager/BG");
                }
                if (bgContainer == null)
                {
                    return;
                }

                if (!TryLoadTexture(_activePath, out var tex) || tex == null)
                {
                    return;
                }

                bool changed = false;

                var rawImages = bgContainer.GetComponentsInChildren<RawImage>(true);
                foreach (var raw in rawImages)
                {
                    if (raw == null) continue;
                    if (!ReferenceEquals(raw.texture, tex))
                    {
                        raw.texture = tex;
                        raw.color = Color.white;
                        changed = true;
                    }
                }

                var images = bgContainer.GetComponentsInChildren<Image>(true);
                if (images != null && images.Length > 0)
                {
                    var sprite = GetOrCreateSprite(_activePath, tex);
                    if (sprite != null)
                    {
                        foreach (var img in images)
                        {
                            if (img == null) continue;
                            if (!ReferenceEquals(img.sprite, sprite))
                            {
                                img.sprite = sprite;
                                img.color = Color.white;
                                changed = true;
                            }
                        }
                    }
                }

                if (changed && !string.Equals(_lastAppliedPath, _activePath, StringComparison.OrdinalIgnoreCase))
                {
                    _lastAppliedPath = _activePath;
                    UnityEngine.Debug.Log($"[CustomBG] 已按 RowIndex 将原生 BG 对象替换为外部图片: [{_activePath}]");
                }
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomBG] RowIndex 原生背景替换失败: {ex.Message}");
            }
        }

        private static void PreloadTexture(string path)
        {
            TryLoadTexture(path, out _);
        }

        private static void PreloadSprite(string path)
        {
            if (TryLoadTexture(path, out var tex) && tex != null)
            {
                GetOrCreateSprite(path, tex);
            }
        }

        private static Sprite GetOrCreateSprite(string path, Texture2D tex)
        {
            if (string.IsNullOrWhiteSpace(path) || tex == null) return null;
            if (SpriteCache.TryGetValue(path, out var sprite) && sprite != null)
            {
                return sprite;
            }

            sprite = Sprite.Create(tex, new Rect(0, 0, tex.width, tex.height), new Vector2(0.5f, 0.5f));
            sprite.name = Path.GetFileNameWithoutExtension(path) + "_extbg_sprite";
            SpriteCache[path] = sprite;
            return sprite;
        }

        private static bool TryLoadTexture(string pathOrRelativePath, out Texture2D texture)
        {
            texture = null;
            if (string.IsNullOrWhiteSpace(pathOrRelativePath)) return false;

            string cacheKey = NormalizePath(pathOrRelativePath);
            if (string.IsNullOrEmpty(cacheKey)) return false;

            if (TextureCache.TryGetValue(cacheKey, out texture) && texture != null)
            {
                return true;
            }

            foreach (var candidate in BuildCandidatePaths(cacheKey))
            {
                string fullPath = Path.IsPathRooted(candidate) ? candidate : Path.Combine(Paths.PluginPath, candidate);
                if (!File.Exists(fullPath))
                {
                    if (MissingPathLogged.Add(fullPath))
                    {
                        UnityEngine.Debug.LogWarning($"[CustomBG] 未找到外部背景: {fullPath}");
                    }
                    continue;
                }

                try
                {
                    byte[] data = File.ReadAllBytes(fullPath);
                    Texture2D tex = new Texture2D(2, 2, TextureFormat.RGBA32, false);
                    if (!ImageConversion.LoadImage(tex, data))
                    {
                        continue;
                    }

                    tex.name = Path.GetFileNameWithoutExtension(fullPath) + "_extbg";
                    TextureCache[cacheKey] = tex;
                    texture = tex;
                    UnityEngine.Debug.Log($"[CustomBG] 已加载外部背景: {fullPath}");
                    return true;
                }
                catch (Exception ex)
                {
                    UnityEngine.Debug.LogError($"[CustomBG] 读取外部背景失败: {fullPath} | {ex.Message}");
                    return false;
                }
            }

            return false;
        }

        private static IEnumerable<string> BuildCandidatePaths(string path)
        {
            if (string.IsNullOrWhiteSpace(path))
            {
                yield break;
            }

            yield return path;

            if (!Path.HasExtension(path))
            {
                foreach (var ext in ExtraExtensions)
                {
                    yield return path + ext;
                }
            }

            string normalized = path.Trim().TrimStart('\\', '/').Replace('/', '\\');
            if (!string.IsNullOrEmpty(normalized))
            {
                string shiyiAssetsPath = Path.Combine("ShiyiAssets_List", normalized);
                yield return shiyiAssetsPath;

                if (!Path.HasExtension(normalized))
                {
                    foreach (var ext in ExtraExtensions)
                    {
                        yield return shiyiAssetsPath + ext;
                    }
                }

                string picturePath = Path.Combine("ShiyiAssets_List", "Picture", normalized);
                yield return picturePath;

                if (!Path.HasExtension(normalized))
                {
                    foreach (var ext in ExtraExtensions)
                    {
                        yield return picturePath + ext;
                    }
                }
            }
        }

        private static string NormalizePath(string path)
        {
            if (string.IsNullOrWhiteSpace(path)) return null;

            string normalized = path.Trim().Trim('"');
            if (Path.IsPathRooted(normalized))
            {
                return normalized.Replace('/', '\\');
            }

            return normalized.TrimStart('\\', '/').Replace('/', '\\');
        }
    }

    [HarmonyPatch(typeof(StringGrid), nameof(StringGrid.Rows), MethodType.Getter)]
    public class UtageStringGrid_Rows_Patch
    {
        private static HashSet<IntPtr> processedGrids = new HashSet<IntPtr>();

        [HarmonyPrefix]
        public static void Prefix(StringGrid __instance)
        {
            if (__instance == null || __instance.Pointer == IntPtr.Zero) return;

            IntPtr gridPtr = __instance.Pointer;
            if (processedGrids.Contains(gridPtr)) return;

            string sheetName = __instance.SheetName;
            if (string.IsNullOrEmpty(sheetName)) return;

            if (Plugin.CustomSheetsCache.TryGetValue(sheetName, out var customRows))
            {
                processedGrids.Add(gridPtr);
                UnityEngine.Debug.Log($"[Mod-Debug] 检测到剧情表 [{sheetName}]，开始注入自定义行数据...");

                try
                {
                    if (__instance.rows != null)
                    {
                        __instance.rows.Clear();
                    }
                    else
                    {
                        __instance.rows = new Il2CppSystem.Collections.Generic.List<StringGridRow>();
                    }

                    foreach (var rowData in customRows)
                    {
                        var il2cppArray = new Il2CppInterop.Runtime.InteropTypes.Arrays.Il2CppStringArray(rowData);
                        __instance.AddRow(il2cppArray);
                    }

                    __instance.ParseHeader();
                    UnityEngine.Debug.Log($"[Mod-Debug] 剧情表 [{sheetName}] 注入成功，新总行数: {__instance.rows.Count}");
                }
                catch (Exception ex)
                {
                    UnityEngine.Debug.LogError($"[Mod-Debug] 动态注入行数据时发生异常: {ex.Message}");
                }
            }
        }
    }

    public static class CustomVoicePatcher
    {
        public static bool TryReplaceVoice(SoundManager soundManager, string characterLabel, AssetFile assetFile, bool isLoop, float volume, float fadeInTime, float fadeOutTime = 0f, bool useEngineDefaultVolume = false)
        {
            try
            {
                if (soundManager == null) return true;

                if (!CustomVoiceManager.TryGetClip(characterLabel, assetFile, out var clip) || clip == null)
                {
                    return true;
                }

                float finalVolume = useEngineDefaultVolume || volume < 0f ? soundManager.VoiceVolume : volume;
                if (finalVolume <= 0f) finalVolume = soundManager.VoiceVolume;
                if (fadeOutTime < 0f) fadeOutTime = 0f;

                soundManager.PlayVoice(characterLabel, clip, isLoop, finalVolume, fadeInTime);
                UnityEngine.Debug.Log($"[CustomVoice] 已替换语音 -> 角色: [{characterLabel ?? "<null>"}] 音量: {finalVolume} 淡入: {fadeInTime}");
                return false;
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[CustomVoice] 替换语音失败: {ex.Message}");
                return true;
            }
        }
    }

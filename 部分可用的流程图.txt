namespace UtageCsvInjectorPlugin
{
    [BepInPlugin("com.shiyimv.utage_csv_injector", "Utage CSV Script Injector", "1.7.0")]
    public class Plugin : BasePlugin 
    {
        public static Dictionary<string, List<string[]>> CustomSheetsCache = new Dictionary<string, List<string[]>>();
        public static string DumpFilePath;

        public override void Load() 
        {
            System.Console.OutputEncoding = System.Text.Encoding.UTF8;

            System.Console.InputEncoding = System.Text.Encoding.UTF8;
            DumpFilePath = Path.Combine(Paths.PluginPath, "ScenarioDump.csv");
            LoadScenarioDump();
            
            Harmony harmony = new Harmony("com.shiyimv.utage_csv_injector");
            harmony.PatchAll();
        }

        public static void LoadScenarioDump()
        {
            try
            {
                if (!File.Exists(DumpFilePath)) return;
                CustomSheetsCache.Clear();
                
                string currentSheetName = null;
                List<string[]> currentRows = new List<string[]>();
                string[] lines = File.ReadAllLines(DumpFilePath, Encoding.UTF8);

                foreach (var line in lines)
                {
                    if (string.IsNullOrEmpty(line) || line.Trim().Length == 0) continue;

                    if (line.StartsWith("#SHEET_NAME", StringComparison.OrdinalIgnoreCase))
                    {
                        if (!string.IsNullOrEmpty(currentSheetName) && currentRows.Count > 0)
                        {
                            CustomSheetsCache[currentSheetName] = new List<string[]>(currentRows);
                        }

                        string[] parts = line.Split(',');
                        if (parts.Length > 1)
                        {
                            currentSheetName = parts[1].Trim();
                            currentRows.Clear();
                        }
                        continue;
                    }

                    if (!string.IsNullOrEmpty(currentSheetName))
                    {
                        string[] cells = line.Split(',');
                        currentRows.Add(cells);
                    }
                }

                if (!string.IsNullOrEmpty(currentSheetName) && currentRows.Count > 0)
                {
                    CustomSheetsCache[currentSheetName] = new List<string[]>(currentRows);
                }
                UnityEngine.Debug.Log($"[Mod] 成功载入自定义剧本，共缓存了 {CustomSheetsCache.Count} 张工作表。");
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[Mod] 载入剧本文件失败: {ex.Message}");
            }
        }
    }

    // ==================== 核心一：长剧本 Rows 内存重定向注入器 ====================
    
    [HarmonyPatch(typeof(StringGrid), nameof(StringGrid.Rows), MethodType.Getter)]
    public class UtageStringGrid_Rows_Patch
    {
        private static HashSet<IntPtr> processedGrids = new HashSet<IntPtr>();

        [HarmonyPrefix]
        public static void Prefix(StringGrid __instance)
        {
            if (__instance == null || __instance.Pointer == IntPtr.Zero) return;

            IntPtr gridPtr = __instance.Pointer;
            if (processedGrids.Contains(gridPtr)) return;

            string sheetName = __instance.SheetName;
            if (string.IsNullOrEmpty(sheetName)) return;

            if (Plugin.CustomSheetsCache.TryGetValue(sheetName, out var customRows))
            {
                processedGrids.Add(gridPtr);
                try
                {
                    if (__instance.rows != null) __instance.rows.Clear();
                    else __instance.rows = new Il2CppSystem.Collections.Generic.List<StringGridRow>();

                    foreach (var rowData in customRows)
                    {
                        var il2cppArray = new Il2CppInterop.Runtime.InteropTypes.Arrays.Il2CppStringArray(rowData);
                        __instance.AddRow(il2cppArray);
                    }
                    __instance.ParseHeader();
                    UnityEngine.Debug.Log($"[Mod] 🎯 剧本表 [{sheetName}] 内存大长篇注入成功！");
                }
                catch (Exception ex) { UnityEngine.Debug.LogError($"[Mod] 注入异常: {ex.Message}"); }
            }
        }
    }

    // ==================== 核心二：劫持 Init 初始化网络，动态追加独立节点 ====================
    
    [HarmonyPatch(typeof(UI_PlotMap), nameof(UI_PlotMap.RefreshUI))]
    public class UI_PlotMap_RefreshUI_Patch
    {
        private static bool hasInjected = false;

        [HarmonyPostfix]
        public static void Postfix(UI_PlotMap __instance)
        {
            if (__instance == null || hasInjected) return;

            try
            {
                // 1. 定位 UI 上的 Level1 按钮容器
                Transform level1 = __instance.transform.Find("Scroll View/Viewport/Content/Buttons/Level1");
                if (level1 == null || level1.childCount == 0) return;

                // 2. 抓取第一个原版按钮作为克隆模板
                GameObject templateGo = level1.GetChild(0).gameObject;
                
                UnityEngine.Debug.LogWarning("[Mod] 正在通过【物理克隆 + Init劫持】生成 Mod 独立剧情节点...");

                // 3. 克隆一个全新的按钮物体，重命名为你的 Mod 分支名
                GameObject myNewButton = GameObject.Instantiate(templateGo, level1);
                myNewButton.name = "[Mod] 完美隐居";

                // 4. 平移位置（防止跟原版重叠，往右移动 350 像素）
                RectTransform rect = myNewButton.GetComponent<RectTransform>();
                if (rect != null)
                {
                    Vector2 pos = rect.anchoredPosition;
                    rect.anchoredPosition = new Vector2(pos.x + 350f, pos.y);
                }

                // 5. 拿到物体身上挂载的原版核心组件 UI_PlotChapterElement
                var chapterElement = myNewButton.GetComponent<UI_PlotChapterElement>();
                if (chapterElement != null)
                {
                    // 强行把它的标签名字修改为你在 CSV 剧本里新加的解锁标签（对应 ,UnlockPlotMap,map_shiyi_1）
                    chapterElement.tagName = "map_shiyi_1"; 
                    
                    // 修改解密后的显示文本
                    if (chapterElement.text_unlock != null)
                    {
                        chapterElement.text_unlock.text = "完美隐居";
                    }

                    // 6. 【精髓所在】：强行调用它原版的初始化函数，让克隆出来的按钮自己去跑原版逻辑
                    chapterElement.Init();
                    chapterElement.InitState();
                }

                myNewButton.SetActive(true);
                hasInjected = true; // 锁定挂载，防止重复实例化

                UnityEngine.Debug.LogWarning("[Mod] 🎯 独立新节点按钮 [完美隐居] 已成功挂载入原版流程图，并安全完成 Init 初始化操作！");
            }
            catch (Exception ex)
            {
                UnityEngine.Debug.LogError($"[Mod] 劫持初始化失败: {ex.Message}");
            }
        }
    }
